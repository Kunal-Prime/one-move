        // Create floating particles
        function createParticles() {
            const container = document.getElementById('particles');
            for (let i = 0; i < 30; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = Math.random() * 100 + '%';
                particle.style.top = Math.random() * 100 + '%';
                particle.style.setProperty('--tx', (Math.random() - 0.5) * 200 + 'px');
                particle.style.setProperty('--ty', (Math.random() - 0.5) * 200 + 'px');
                particle.style.animationDelay = Math.random() * 15 + 's';
                container.appendChild(particle);
            }
        }

        const observer = new IntersectionObserver((entries) => {

        // AI-POWERED CLARITY GENERATION
        async function generateClarity() {
            const input = document.getElementById('thoughtInput').value.trim();
            const btn = document.getElementById('generateBtn');
            const outputCard = document.getElementById('outputCard');
            const errorMsg = document.getElementById('errorMessage');

            // Hide previous errors
            errorMsg.classList.remove('show');

            if (!input) {
                showError('Please share what\'s on your mind first');
                return;
            }

            if (input.length < 20) {
                showError('Please provide more context (at least 20 characters)');
                return;
            }

            // Show loading state
            btn.disabled = true;
            btn.classList.add('loading');
            btn.textContent = '🧠 AI is thinking...';

            try {
                // Call Claude API (built-in to claude.ai artifacts)
                const response = await fetch("https://api.anthropic.com/v1/messages", {
                    method: "POST",
                    headers: {
                        "Content-Type": "application/json",
                    },
                    body: JSON.stringify({
                        model: "claude-sonnet-4-20250514",
                        max_tokens: 1000,
                        messages: [{
                            role: "user",
                            content: `You are OneMove, a decision-compression tool that helps people break through analysis paralysis.

USER'S MENTAL CHAOS:
${input}

YOUR JOB: Extract clarity and force action. Respond ONLY in this exact JSON format:

{
  "coreProblem": "One clear sentence identifying what's actually blocking them",
  "canControl": ["Item 1", "Item 2", "Item 3"],
  "cannotControl": ["Item 1", "Item 2"],
  "nextMove": "ONE specific, small action they can complete in the next 60 minutes. Must be concrete and time-boxed."
}

RULES:
- Be direct but empathetic
- Core problem should reveal the real blocker, not just restate their words
- "Can control" should be immediate, specific actions within their power
- "Cannot control" should acknowledge external factors they're worrying about
- Next move must be SMALL (completable in <60 min), SPECIFIC (clear what to do), IMMEDIATE (can start right now)
- No vague advice like "just start" or "stay positive"
- Total response under 150 words

Example good next move: "Open Google Docs, write just the introduction paragraph of the shortest assignment. Set timer for 15 minutes."
Example bad next move: "Start working on your assignments" (too vague)

Respond ONLY with valid JSON, no other text.`
                        }]
                    })
                });

                if (!response.ok) {
                    throw new Error(`API Error: ${response.status}`);
                }

                const data = await response.json();
                const aiText = data.content[0].text;
                
                // Parse AI response
                let result;
                try {
                    // Clean the response (remove markdown code blocks if present)
                    const cleanText = aiText.replace(/```json\n?/g, '').replace(/```\n?/g, '').trim();
                    result = JSON.parse(cleanText);
                } catch (e) {
                    // If JSON parsing fails, try to extract structured data from text
                    result = parseTextResponse(aiText);
                }

                // Validate response
                if (!result.coreProblem || !result.nextMove) {
                    throw new Error('Invalid AI response format');
                }

                // Display results
                displayResults(result);

                // Save session if requested
                const rememberSession = document.getElementById('rememberSession').checked;
                if (rememberSession) {
                    saveSession(input, result);
                }

            } catch (error) {
                console.error('Error:', error);
                showError('Something went wrong. Please try again or rephrase your thoughts.');
            } finally {
                // Reset button
                btn.disabled = false;
                btn.classList.remove('loading');
                btn.textContent = '🧠 Generate My Next Move';
            }
        }

        function parseTextResponse(text) {
            // Fallback parser if AI doesn't return JSON
            const lines = text.split('\n').filter(l => l.trim());
            return {
                coreProblem: lines[0] || "You're experiencing decision paralysis.",
                canControl: ["The next hour", "Your immediate actions", "What you focus on now"],
                cannotControl: ["Past decisions", "Other people's reactions", "External circumstances"],
                nextMove: lines[lines.length - 1] || "Take one small step forward right now."
            };
        }

        function displayResults(result) {
            const outputCard = document.getElementById('outputCard');
            
            // Populate core problem
            document.getElementById('coreProblem').textContent = result.coreProblem;
            
            // Populate control lists
            const controlList = document.getElementById('controlList');
            controlList.innerHTML = '';
