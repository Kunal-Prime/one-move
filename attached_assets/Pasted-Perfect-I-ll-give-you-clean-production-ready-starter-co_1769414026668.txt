## 5️⃣ `decisionEngine.js` (THE AI BRAIN)

This is the **core logic**. Read it carefully.

```js
import fetch from "node-fetch";

const GEMINI_API_URL =
  "https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent";

export async function generateOneMove(userInput) {
  const prompt = `
You are a constrained decision engine called OneMove.

RULES (NON-NEGOTIABLE):
- Do NOT ask follow-up questions.
- Do NOT provide multiple actions.
- Do NOT sound motivational or emotional.
- Do NOT exceed the structure below.
- The action MUST be doable within 60 minutes.

TASK:
Convert the user's messy thoughts into structured clarity.

OUTPUT FORMAT (EXACT):
🧠 Core problem:
<one sentence>

✅ What you control:
<one short bullet>

❌ What you don’t control:
<one short bullet>

⚡ OneMove (next 60 minutes):
<one concrete action>

USER THOUGHTS:
"""${userInput}"""
`;

  const response = await fetch(
    `${GEMINI_API_URL}?key=${process.env.GEMINI_API_KEY}`,
    {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        contents: [{ parts: [{ text: prompt }] }],
      }),
    }
  );

  const data = await response.json();

  if (!data.candidates) {
    throw new Error("AI failed to generate response");
  }

  return data.candidates[0].content.parts[0].text;
}
```

## 6️⃣ `server.js` (API layer)

```js
import express from "express";
import cors from "cors";
import dotenv from "dotenv";
import { generateOneMove } from "./decisionEngine.js";

dotenv.config();

const app = express();
app.use(cors());
app.use(express.json());

app.post("/onemove", async (req, res) => {
  try {
    const { thoughts } = req.body;

    if (!thoughts || thoughts.length < 10) {
      return res.status(400).json({
        error: "Please provide more detailed thoughts.",
      });
    }

    const result = await generateOneMove(thoughts);

    res.json({ result });
  } catch (err) {
    res.status(500).json({
      error: "Something went wrong generating your OneMove.",
    });
  }
});

app.listen(3000, () => {
  console.log("🧠 OneMove AI Engine running on port 3000");
});
```
