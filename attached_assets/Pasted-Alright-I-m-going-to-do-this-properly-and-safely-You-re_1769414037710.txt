## 4️⃣ `db.js` — SQLite (safe, simple, fast)

```js
import Database from "better-sqlite3";

const db = new Database("onemove.db");

// Create table if not exists
db.prepare(`
  CREATE TABLE IF NOT EXISTS history (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    thoughts TEXT NOT NULL,
    result TEXT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
  )
`).run();

export function saveHistory(thoughts, result) {
  const stmt = db.prepare(
    "INSERT INTO history (thoughts, result) VALUES (?, ?)"
  );
  stmt.run(thoughts, result);
}

export function getHistory(limit = 10) {
  return db
    .prepare(
      "SELECT id, thoughts, result, created_at FROM history ORDER BY created_at DESC LIMIT ?"
    )
    .all(limit);
}
```

export function isValidOneMoveOutput(text) {
  const requiredSections = [
    "🧠 Core problem:",
    "✅ What you control:",
    "❌ What you don’t control:",
    "⚡ OneMove"
  ];

  return requiredSections.every(section => text.includes(section));
}

## 6️⃣ `decisionEngine.js` — AI Brain (hardened)

```js
import fetch from "node-fetch";
import { isValidOneMoveOutput } from "./validator.js";

const API_URL =
  "https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent";

export async function generateOneMove(thoughts) {
  const prompt = `
You are OneMove — a constrained decision engine.

RULES:
- No follow-up questions
- No motivation or empathy
- No multiple actions
- One action must be doable in 60 minutes
- Output must follow EXACT structure

FORMAT:
🧠 Core problem:
<one sentence>

✅ What you control:
<one bullet>

❌ What you don’t control:
<one bullet>

⚡ OneMove (next 60 minutes):
<one concrete action>

USER INPUT:
"""${thoughts}"""
`;

  const controller = new AbortController();
  setTimeout(() => controller.abort(), 10000); // 10s timeout

  const res = await fetch(`${API_URL}?key=${process.env.GEMINI_API_KEY}`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      contents: [{ parts: [{ text: prompt }] }]
    }),
    signal: controller.signal
  });

  const data = await res.json();
  const output = data?.candidates?.[0]?.content?.parts?.[0]?.text;

  if (!output || !isValidOneMoveOutput(output)) {
    throw new Error("Invalid AI output");
  }

  return output;
}
```

## 7️⃣ `server.js` — API with history support

```js
import express from "express";
import cors from "cors";
import dotenv from "dotenv";
import { generateOneMove } from "./decisionEngine.js";
import { saveHistory, getHistory } from "./db.js";

dotenv.config();

const app = express();
app.use(cors());
app.use(express.json());

// Generate OneMove
app.post("/onemove", async (req, res) => {
  try {
    const { thoughts, save } = req.body;

    if (!thoughts || thoughts.trim().length < 15) {
      return res.status(400).json({
        error: "Please write a bit more so OneMove can help."
      });
    }

    const result = await generateOneMove(thoughts);

    // Save only if user explicitly allows
    if (save === true) {
      saveHistory(thoughts, result);
    }

    res.json({ result });
  } catch (err) {
    res.status(500).json({
      error: "OneMove failed to generate clarity. Try again."
    });
  }
});

// Fetch history (user-initiated only)
app.get("/history", (req, res) => {
  try {
    const limit = Number(req.query.limit) || 10;
    const history = getHistory(limit);
    res.json({ history });
  } catch {
    res.status(500).json({ error: "Could not fetch history." });
  }
});

app.listen(process.env.PORT, () => {
  console.log(`🧠 OneMove running on port ${process.env.PORT}`);
});
```
